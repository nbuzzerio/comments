SONGS
-------
CREATE TABLE sdc_comments.songs
(
    song_id integer NOT NULL,
    CONSTRAINT songs_pkey PRIMARY KEY (song_id)
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

ALTER TABLE sdc_comments.songs
    OWNER to postgres;




--------------------------------------------------------------------------



COMMENTS
---------
CREATE TABLE sdc_comments.comments
(
    comment_id integer NOT NULL,
    song_id integer,
    user_id integer,
    user_icon character varying(255) COLLATE pg_catalog."default",
    user_name character varying(255) COLLATE pg_catalog."default",
    message text COLLATE pg_catalog."default",
    audio_position integer,
    posted_at character varying(255) COLLATE pg_catalog."default",
    "createdAt" timestamp with time zone,
    "updatedAt" timestamp with time zone,
    CONSTRAINT comments_pkey PRIMARY KEY (comment_id),
    CONSTRAINT song_id FOREIGN KEY (song_id)
        REFERENCES sdc_comments.songs (song_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

ALTER TABLE sdc_comments.comments
    OWNER to postgres;
-- Index: fki_song_id

-- DROP INDEX sdc_comments.fki_song_id;

CREATE INDEX fki_song_id
    ON sdc_comments.comments USING btree
    (song_id ASC NULLS LAST)
    TABLESPACE pg_default;
